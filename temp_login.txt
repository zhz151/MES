@page "/login"
@using MES.Blazor.Models
@using MES.Blazor.Services
@using MudBlazor
@inject IAuthService AuthService
@inject NavigationManager Navigation
@inject HttpClient HttpClient

<MudContainer MaxWidth="MaxWidth.Small" Class="d-flex justify-center align-center" Style="min-height: 100vh;">
    <MudCard Class="pa-8" Elevation="3">
        <MudCardContent>
            <MudText Typo="Typo.h4" Align="Align.Center" Class="mb-4" Color="Color.Primary">
                MES 生产执行系统
            </MudText>

            <MudText Typo="Typo.subtitle2" Align="Align.Center" Class="mb-6" Color="Color.Secondary">
                不锈钢无缝钢管 MES 系统
            </MudText>

            @if (_showError)
            {
                <MudAlert Severity="Severity.Error" Class="mb-4" Dense="true">
                    @_errorMessage
                </MudAlert>
            }

            <MudTextField @bind-Value="_model.Username"
                          Label="用户名 / 邮箱"
                          Variant="Variant.Outlined"
                          Margin="Margin.Normal"
                          Adornment="Adornment.Start"
                          AdornmentIcon="@Icons.Material.Filled.Person"
                          Disabled="@_isLoading">
            </MudTextField>

            <MudTextField @bind-Value="_model.Password"
                          Label="密码"
                          Variant="Variant.Outlined"
                          Margin="Margin.Normal"
                          Type="InputType.Password"
                          Adornment="Adornment.Start"
                          AdornmentIcon="@Icons.Material.Filled.Lock"
                          Disabled="@_isLoading">
            </MudTextField>

            <MudButton Variant="Variant.Filled"
                       Color="Color.Primary"
                       Size="Size.Large"
                       FullWidth="true"
                       Class="mt-4"
                       OnClick="OnLoginAsync"
                       Disabled="@_isLoading">
                @if (_isLoading)
                {
                    <MudText>登录中...</MudText>
                }
                else
                {
                    <MudText>登录系统</MudText>
                }
            </MudButton>

            <MudButton Variant="Variant.Outlined"
                       Color="Color.Secondary"
                       FullWidth="true"
                       Class="mt-2"
                       OnClick="OnTestApi"
                       Disabled="@_isLoading">
                <MudText>测试API连接</MudText>
            </MudButton>

            <MudPaper Class="mt-6 pa-4" Elevation="0" Style="background-color: #e3f2fd;">
                <MudText Typo="Typo.caption" Color="Color.Default">
                    <strong>测试账号：</strong><br/>
                    邮箱: admin@mes.com<br/>
                    密码: Admin@123456
                </MudText>
            </MudPaper>

            <MudPaper Class="mt-4 pa-4" Elevation="0" Style="background-color: #fff9c4;">
                <MudText Typo="Typo.caption" Color="Color.Default">
                    <strong>诊断信息：</strong><br/>
                    API Base: @_apiBaseUrl<br/>
                    Login URL: @_loginUrl
                </MudText>
            </MudPaper>
        </MudCardContent>
    </MudCard>
</MudContainer>

@code {
    private LoginRequest _model = new();
    private bool _isLoading = false;
    private bool _showError = false;
    private string _errorMessage = string.Empty;
    private string _apiBaseUrl = string.Empty;
    private string _loginUrl = string.Empty;

    protected override void OnInitialized()
    {
        _apiBaseUrl = HttpClient.BaseAddress?.ToString() ?? "Unknown";
        _loginUrl = $"{_apiBaseUrl}api/Auth/login";
        Console.WriteLine($"[Login] API Base URL: {_apiBaseUrl}");
        Console.WriteLine($"[Login] Login URL: {_loginUrl}");

        // 清除旧的token用于测试
        Console.WriteLine("[Login] Clearing old tokens from localStorage...");
        _ = JSRuntime.InvokeVoidAsync("localStorage.removeItem", "auth_token");
        _ = JSRuntime.InvokeVoidAsync("localStorage.removeItem", "user_info");
    }

    [Inject]
    private IJSRuntime JSRuntime { get; set; } = default!;

    private async Task OnLoginAsync()
    {
        Console.WriteLine("[Login] ========== Login Button Clicked ==========");
        _isLoading = true;
        _showError = false;
        StateHasChanged();

        try
        {
            Console.WriteLine($"[Login] Attempting login for: {_model.Username}");
            Console.WriteLine($"[Login] API URL: {_loginUrl}");
            Console.WriteLine($"[Login] Request Body: {System.Text.Json.JsonSerializer.Serialize(_model)}");

            var result = await AuthService.LoginAsync(_model);
            Console.WriteLine($"[Login] Result - Success: {result.Success}, Message: {result.Message}");

            if (result.Success)
            {
                Console.WriteLine("[Login] Login successful, redirecting to home...");
                Console.WriteLine("[Login] Using Navigation.NavigateTo with forceLoad=false");
                Navigation.NavigateTo("/", forceLoad: false);

                // 添加延迟后手动刷新，确保认证状态更新
                await Task.Delay(100);
                Console.WriteLine("[Login] Triggering page refresh...");
                await JSRuntime.InvokeVoidAsync("eval", "location.reload();");
            }
            else
            {
                Console.WriteLine($"[Login] Login failed: {result.Message}");
                _errorMessage = result.Message;
                _showError = true;
                StateHasChanged();
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine($"[Login] Exception: {ex}");
            _errorMessage = $"登录失败: {ex.Message}";
            _showError = true;
            StateHasChanged();
        }
        finally
        {
            _isLoading = false;
            Console.WriteLine("[Login] ========== Login Process Completed ==========");
            StateHasChanged();
        }
    }

    private async Task OnTestApi()
    {
        Console.WriteLine("[Login] ========== Testing API Connection ==========");
        _errorMessage = string.Empty;
        _showError = false;
        StateHasChanged();

        try
        {
            Console.WriteLine($"[Login] Testing API at: {_apiBaseUrl}api/DbAdmin/health");

            var response = await HttpClient.GetAsync($"{_apiBaseUrl}api/DbAdmin/health");
            Console.WriteLine($"[Login] Health Status: {response.StatusCode}");

            var content = await response.Content.ReadAsStringAsync();
            Console.WriteLine($"[Login] Health Response: {content}");

            if (response.IsSuccessStatusCode)
            {
                _errorMessage = $"API连接正常！Status: {response.StatusCode}";
                _showError = true;
            }
            else
            {
                _errorMessage = $"API连接失败！Status: {response.StatusCode}\n响应: {content}";
                _showError = true;
            }
        }
        catch (Exception ex)
        {
            Console.WriteLine($"[Login] API Test Exception: {ex}");
            _errorMessage = $"API连接异常: {ex.Message}";
            _showError = true;
        }

        StateHasChanged();
    }
}
